// Bütün data-page atributu olan düymələrə klik dinləyicisi əlavə olunur document.querySelectorAll('button[data-page]').forEach(button => { button.addEventListener('click', () => { const page = button.getAttribute('data-page'); if (page) { // iframe src-ni dəyişir document.getElementById('contentFrame').src = page; } }); }); // Əlavə: data-frame atributu olan düymələr üçün də eyni funksionallıq document.querySelectorAll('button[data-frame]').forEach(button => { button.addEventListener('click', () => { const frameUrl = button.getAttribute('data-frame'); if (frameUrl) { document.getElementById('contentFrame').src = frameUrl; } }); });
